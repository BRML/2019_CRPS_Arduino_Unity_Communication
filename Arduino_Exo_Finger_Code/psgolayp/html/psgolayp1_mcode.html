1function [ SG0 ] = psgolayp ( A, k, n )
 2%psgolay compute x, x', x" from data A with k-polynomial over 
 3%   the preceding 2*n+1 points.
 4%   Differs from standard sgolay in that it introduces NO delay
 5%   2*n+1 is the number of past points that are used in the filter
 6%   k is the order of the polynomial 
 7%
 8%   Example use:
 9%   T=0.05;t=(0:99)*T; s = sin(2*pi*t); s_noise = s + 0.5*rand(size(t));
10%   plot(s_noise); hold on; d=psgolayp(s_noise,2,20);plot(d,'g'); hold off;
11
12
13%if round(n) ~= n 
14%    error(generatemsgid('MustBeInteger'),'Number of data points must be an integer.'), end
15%if round(k) ~= k
16%    error(generatemsgid('MustBeInteger'),'Polynomial order must be an integer.'), end
17%if k > n-1, error(generatemsgid('InvalidRange'),'It is silly to have a polynomial degree higher than # data.'), end
18
19% compute the SG parameters for the last item.  Based on code from
20% Diederick C. Niehorster <dcniehorster@hku.hk>.  Lots of magic ;-)
21x = [-n:n]';
22df = cumprod([ones(2*n+1,1) x*ones(1,k)],2) \ eye(2*n+1);
23df = df.';
24SG0 = zeros(10,1);
25
26hx = 1;
27for i=1:k
28    hx = [hx [n].^i];
29end
30
31% these are the SG filter coefficients
32fc = df*hx';
33
34% now do the smoothing!
35for i = 2*n+1:length(A)  
36    SG0(i) = dot(fc, A(i-2*n:i));
37end
38
39end
